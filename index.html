<body>

  <div id="controls">

    <label for="structureInput">Search structure filename:</label>

    <input type="text" id="structureInput" placeholder="e.g. Q6ZV65_O14744_Boltz-2_v2-1-1" />

  </div>

  <div id="viewer"></div>


  <script>

    const viewer = new Molstar.Viewer('viewer', {

      layoutIsExpanded: true,

      showImportControls: false,

    });


    const input = document.getElementById('structureInput');

    input.addEventListener('change', () => {

      const filename = input.value.trim();

      if (!filename.endsWith('.cif')) {

        alert("Please enter a valid CIF filename (with .cif extension)");

        return;

      }

      const url = `https://raw.githubusercontent.com/structural-interactome/human-interactome/main/interaction_structures/${filename}`;

      viewer.loadStructureFromUrl(url, 'mmcif')

        .catch(err => alert(`Failed to load file: ${err}`));

    });

  </script>

</body>
